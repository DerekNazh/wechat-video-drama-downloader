# 前端响应式状态管理架构改进

> 解决问题：视频状态改变时，当前 Tab 的 UI 不能实时响应，需要在多个组件回调中添加逻辑。

---

## 一、当前问题分析

### 1.1 问题场景

```
用户在"短视频" Tab
↓
直播回放下载完成（SSE 推送 task_completed）
↓
作者统计更新了（全局）
↓
但"直播回放" Tab 的统计数字没有实时更新
↓
用户需要手动切换 Tab 才能看到最新状态
```

### 1.2 根本原因

**当前架构：**
```
SSE 事件 → handleSSETaskCompleted() → 直接操作 DOM
                                       ↓
                                   分散在多个文件中
                                   authorDetail.js, videoListRender.js, status.js...
```

**问题：**
1. 状态更新逻辑分散在多个组件中
2. 每个组件需要自己判断"是否需要更新"
3. Tab 切换后，旧 Tab 的状态更新被忽略
4. 新增状态变化点时，需要修改多处代码

---

## 二、架构改进方案

### 2.1 核心思想：单一数据源 + 响应式渲染

```
┌─────────────────────────────────────────────────────────────────┐
│                         SSE 事件                                 │
│  task_progress, task_completed, import_progress                 │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                    State 层（单一数据源）                         │
│                                                                 │
│  State.videos[video_id] = {                                     │
│      downloaded: true,                                          │
│      download_path: "...",                                      │
│      status: "done",                                            │
│      video_type: "short_video" | "live_replay"                 │
│  }                                                              │
│                                                                 │
│  State.tasks[task_id] = { ... }                                │
│  State.authors[author_id] = { stats: { ... } }                 │
│                                                                 │
│  State.emit('videos:update', { videoId, changes })             │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                    响应式渲染层                                   │
│                                                                 │
│  // 订阅 State 变化，自动重新计算需要渲染的内容                    │
│  State.on('videos:update', (e) => {                             │
│      // 自动判断：这个视频是否在当前 Tab 的过滤范围内？            │
│      // 如果是，更新 DOM；如果不是，只更新 State                  │
│      ReactiveRenderer.onVideoUpdate(e);                        │
│  });                                                            │
│                                                                 │
│  // Tab 统计自动更新                                             │
│  State.on('videos:update', (e) => {                             │
│      TabStats.update(e.videoId);                               │
│  });                                                            │
└─────────────────────────────────────────────────────────────────┘
```

### 2.2 关键改进点

| 改进点 | 当前行为 | 改进后 |
|--------|----------|--------|
| 状态存储 | 分散在 `_authorVideosData`, `_activeTasks`, `_catalogData` | 统一在 `State.videos`, `State.tasks` |
| 状态更新 | SSE 直接操作 DOM | SSE 只更新 State，触发事件 |
| 渲染触发 | 手动调用 `updateSingleVideoCompleted()` | 自动响应 State 变化 |
| Tab 统计 | 切换时重新请求后端 | 实时计算，无需请求 |
| 过滤判断 | 每个组件自己判断 | 统一在 ReactiveRenderer 中判断 |

## 三、具体实现

### 3.1 扩展 State.videos 支持视频类型

```javascript
// static/js/api/state/tables/videos.js

State.videos = {
  // 现有方法...

  // 新增：按类型获取统计
  getStatsByType: function(username) {
    var videos = this.getAuthorVideos(username);
    var shortVideoTotal = 0, shortVideoDownloaded = 0;
    var replayTotal = 0, replayDownloaded = 0;

    videos.forEach(function(v) {
      var type = v.video_type || 'short_video';
      if (type === 'short_video') {
        shortVideoTotal++;
        if (v.downloaded) shortVideoDownloaded++;
      } else if (type === 'live_replay') {
        replayTotal++;
        if (v.downloaded) replayDownloaded++;
      }
    });

    return {
      short_video: { total: shortVideoTotal, downloaded: shortVideoDownloaded },
      live_replay: { total: replayTotal, downloaded: replayDownloaded },
      total: shortVideoTotal + replayTotal,
      downloaded: shortVideoDownloaded + replayDownloaded
    };
  },

  // 新增：更新单个视频状态并触发事件
  updateStatus: function(videoId, updates) {
    var video = this._findVideo(videoId);
    if (!video) return;

    var oldType = video.video_type || 'short_video';
    var oldDownloaded = video.downloaded;

    Object.assign(video, updates);

    var newType = video.video_type || 'short_video';
    var newDownloaded = video.downloaded;

    // 触发事件，包含变化信息
    State.emit('videos:status', {
      videoId: videoId,
      video: video,
      changes: {
        downloaded: { old: oldDownloaded, new: newDownloaded },
        video_type: { old: oldType, new: newType }
      }
    });
  },

  _findVideo: function(videoId) {
    // 从所有作者的视频中查找
    for (var username in _videos) {
      var authorVideos = _videos[username];
      if (authorVideos.has(videoId)) {
        return authorVideos.get(videoId);
      }
    }
    return null;
  }
};
```

### 3.2 响应式渲染器

```javascript
// static/js/component/reactiveRenderer.js

var ReactiveRenderer = {
  // 当前 Tab 状态
  _currentTab: 'all',           // all | downloading | downloaded | pending
  _currentVideoType: 'short_video', // short_video | live_replay
  _currentAuthor: null,

  // 初始化：订阅 State 事件
  init: function() {
    var self = this;

    // 订阅视频状态变化
    State.on('videos:status', function(e) {
      self.onVideoStatusChange(e);
    });

    // 订阅任务进度变化
    State.on('tasks:update', function(e) {
      self.onTaskProgressChange(e);
    });

    // 订阅任务完成
    State.on('tasks:complete', function(e) {
      self.onTaskComplete(e);
    });
  },

  // 设置当前 Tab（由 Tab 切换调用）
  setCurrentTab: function(tab, videoType) {
    this._currentTab = tab;
    this._currentVideoType = videoType;
    // Tab 切换时，重新渲染当前视图
    this.refreshCurrentView();
  },

  // 视频状态变化处理
  onVideoStatusChange: function(e) {
    var video = e.video;
    var changes = e.changes;

    // 1. 更新 Tab 统计（无论当前在哪个 Tab）
    this._updateTabStats(video);

    // 2. 判断是否需要更新 DOM
    if (!this._shouldRender(video)) {
      // 不在当前视图范围内，只更新 State，不操作 DOM
      console.log('[ReactiveRenderer] 视频不在当前视图，跳过渲染:', video.id);
      return;
    }

    // 3. 更新 DOM
    this._updateVideoRow(video, changes);

    // 4. 如果是从"待下载"变为"已下载"，且当前在"正在下载" Tab，显示 Toast
    if (changes.downloaded && !changes.downloaded.old && changes.downloaded.new) {
      if (this._currentTab === 'downloading') {
        this._showCompletionToast(video);
        // 从列表中移除（带动画）
        this._removeFromDownloadingList(video.id);
      }
    }
  },

  // 判断视频是否在当前视图范围内
  _shouldRender: function(video) {
    var videoType = video.video_type || 'short_video';

    // 类型过滤
    if (videoType !== this._currentVideoType) {
      return false;
    }

    // Tab 过滤
    switch (this._currentTab) {
      case 'downloading':
        return !video.downloaded && this._isTaskActive(video.id);
      case 'downloaded':
        return video.downloaded;
      case 'pending':
        return !video.downloaded && !this._isTaskActive(video.id);
      default: // 'all'
        return true;
    }
  },

  // 更新 Tab 统计
  _updateTabStats: function(video) {
    var videoType = video.video_type || 'short_video';
    var username = video.username || this._currentAuthor?.username;

    if (!username) return;

    // 从 State 实时计算统计
    var stats = State.videos.getStatsByType(username);

    // 更新头部统计
    this._updateHeaderStats(stats);

    // 更新 Tab 标签数字
    this._updateTabLabels(stats);
  },

  // 更新头部统计
  _updateHeaderStats: function(stats) {
    var shortVideoEl = document.getElementById('authorShortVideo');
    var replayEl = document.getElementById('authorReplay');
    var downloadedEl = document.getElementById('authorDownloaded');

    if (shortVideoEl) shortVideoEl.textContent = stats.short_video.total;
    if (replayEl) replayEl.textContent = stats.live_replay.total;

    // downloaded 根据当前类型显示
    var downloadedCount = this._currentVideoType === 'live_replay'
      ? stats.live_replay.downloaded
      : stats.short_video.downloaded;
    if (downloadedEl) downloadedEl.textContent = downloadedCount;
  },

  // 更新 Tab 标签数字
  _updateTabLabels: function(stats) {
    var typeStats = this._currentVideoType === 'live_replay'
      ? stats.live_replay
      : stats.short_video;

    var downloadingCount = State.tasks.getRunning().filter(function(t) {
      var video = State.videos.get(t.video_id);
      return video && (video.video_type || 'short_video') === this._currentVideoType;
    }.bind(this)).length;

    // 更新 Tab 标签
    var tabs = document.querySelectorAll('.video-tab');
    tabs.forEach(function(tab) {
      var tabType = tab.dataset.tab;
      var count = 0;
      switch (tabType) {
        case 'all':
          count = typeStats.total;
          break;
        case 'downloaded':
          count = typeStats.downloaded;
          break;
        case 'pending':
          count = typeStats.total - typeStats.downloaded - downloadingCount;
          break;
        case 'downloading':
          count = downloadingCount;
          break;
      }
      // 更新 Tab 文本（保留原始文本，只更新数字）
      var baseText = tab.dataset.baseText || tab.textContent.replace(/\s*\(\d+\)/, '');
      tab.dataset.baseText = baseText;
      tab.textContent = baseText + ' (' + count + ')';
    });
  },

  // 显示完成 Toast
  _showCompletionToast: function(video) {
    if (typeof showToast === 'function') {
      showToast({
        type: 'success',
        title: '下载完成',
        message: video.title || '视频已下载完成',
        duration: 3000
      });
    }
  },

  // 从"正在下载"列表移除（带动画）
  _removeFromDownloadingList: function(videoId) {
    var row = document.querySelector('.video-row[data-id="' + videoId + '"]');
    if (!row) return;

    // 添加淡出动画
    row.style.transition = 'opacity 0.3s, transform 0.3s';
    row.style.opacity = '0';
    row.style.transform = 'translateX(-20px)';

    setTimeout(function() {
      row.remove();
      // 更新分页
      if (typeof _removeRowAndUpdatePagination === 'function') {
        _removeRowAndUpdatePagination(row);
      }
    }, 300);
  },

  // 刷新当前视图
  refreshCurrentView: function() {
    if (!this._currentAuthor) return;

    var videos = State.videos.getAuthorVideos(this._currentAuthor.username);
    if (typeof renderVideoList === 'function') {
      renderVideoList(videos, this._currentTab);
    }

    // 更新统计
    var stats = State.videos.getStatsByType(this._currentAuthor.username);
    this._updateHeaderStats(stats);
    this._updateTabLabels(stats);
  },

  _isTaskActive: function(videoId) {
    var task = State.tasks.getByVideoId(videoId);
    return task && task.status !== 'done' && task.status !== 'error';
  }
};
```

### 3.3 简化 SSE 处理逻辑

```javascript
// static/js/datahandle/authorDetail.js

// 修改后的 SSE 处理：只更新 State，不直接操作 DOM
function handleSSETaskCompleted(data) {
  var videoId = data.video_id;
  if (!videoId) return;

  // 1. 更新 State.videos
  State.videos.updateStatus(videoId, {
    downloaded: true,
    download_path: data.download_path || ''
  });

  // 2. 更新 State.authors 统计
  State.authors.updateStats(data.username, {
    downloaded: data.downloaded,
    total: data.total
  });

  // 3. 从 State.tasks 移除
  State.tasks.remove(data.task_id);

  // 4. 更新全局统计
  if (data.today_count !== undefined) {
    updateStatusFromPoll({
      today_count: data.today_count,
      today_downloaded: data.today_downloaded,
      total_videos: data.total_videos,
    });
  }

  // 不再需要手动调用 updateSingleVideoCompleted()
  // ReactiveRenderer 会自动响应 State 变化
}

function handleSSETaskProgress(data) {
  var taskId = data.id;
  if (!taskId) return;

  // 1. 更新 State.tasks
  if (data.status === 'done' || data.status === 'error') {
    // 终态：触发完成事件
    if (data.status === 'done') {
      State.emit('tasks:complete', { taskId: taskId, data: data });
    }
    State.tasks.remove(taskId);
  } else {
    // 运行中：更新进度
    State.tasks.update({
      id: taskId,
      video_id: data.video_id,
      status: data.status,
      percent: data.percent,
      downloaded: data.downloaded,
      size: data.total_size,
      speed: data.speed
    });
  }

  // 不再需要手动调用 updateSingleVideoProgress()
  // ReactiveRenderer 会自动响应 State 变化
}
```

### 3.4 Tab 切换简化

```javascript
// static/js/component/videoListRender.js

async function switchVideoType(type) {
  if (_currentVideoType === type) return;

  _currentVideoType = type;

  // 更新 Tab 样式
  document.querySelectorAll('.video-type-tab').forEach(tab => {
    tab.classList.toggle('active', tab.dataset.type === type);
  });

  // 通知 ReactiveRenderer 当前类型
  ReactiveRenderer.setCurrentTab(_currentTab, type);

  // 不再需要重新请求后端
  // ReactiveRenderer 会从 State 中获取数据并渲染
  ReactiveRenderer.refreshCurrentView();
}
```

## 四、数据流对比

### 4.1 改进前

```
SSE: task_completed
    │
    ├──→ handleSSETaskCompleted()
    │        ├──→ _authorVideosData[username].videos[id].downloaded = true
    │        ├──→ updateAuthorCardStats()
    │        ├──→ updateAuthorStats()
    │        ├──→ updateSingleVideoCompleted()  ← 需要判断当前 Tab
    │        ├──→ refreshActiveTasks()
    │        └──→ loadAllDataFromBackend()
    │
    └──→ 如果在其他 Tab，部分更新被忽略
```

### 4.2 改进后

```
SSE: task_completed
    │
    └──→ handleSSETaskCompleted()
             ├──→ State.videos.updateStatus()
             │        └──→ State.emit('videos:status')
             │
             └──→ ReactiveRenderer.onVideoStatusChange()
                      ├──→ _updateTabStats()  ← 所有 Tab 统计都更新
                      ├──→ _shouldRender()    ← 判断是否在当前视图
                      ├──→ _updateVideoRow()  ← 只更新可见的 DOM
                      └──→ _showCompletionToast() ← 显示提示
```

## 五、优势总结

| 方面 | 改进前 | 改进后 |
|------|--------|--------|
| 状态更新 | 分散在多个函数 | 统一在 State 层 |
| DOM 更新 | 手动调用，需要判断 | 自动响应，智能判断 |
| Tab 统计 | 切换时重新请求 | 实时计算，无需请求 |
| 新增状态 | 修改多处代码 | 只需更新 State |
| 代码维护 | 高耦合 | 低耦合，易扩展 |
| UX 体验 | 状态可能不一致 | 实时一致，有反馈 |
