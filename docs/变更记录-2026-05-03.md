## 1. Bug 修复：选中已下载视频点击下载不再重复提交

**文件**: `static/js/datahandle/videoOps.js`

**问题**: `downloadSelected()` 直接把所有选中 video_ids 发送给后端，不检查视频是否已下载，导致已下载视频被重复提交。

**改动**:

```js

  btn.textContent = '添加中...';

    method: 'POST',

    })

    const count = data.data?.count || 0;
    showToast({ type: 'success', title: '添加成功', message: `已将 ${count} 个视频添加到下载队列` });
    // ... 后续操作用 Array.from(_selectedVideos)
  }
}
```

```js
// === 修改后 ===
async function downloadSelected() {
  if (_selectedVideos.size === 0) return;
  if (!_currentAuthor) return;

  // 新增：过滤掉已下载的视频
  const videos = _authorVideosData[_currentAuthor.username]?.videos || {};
  const pendingIds = Array.from(_selectedVideos).filter(id => {
    const video = videos[id];
    return video && !video.downloaded;
  });
  const alreadyDownloaded = _selectedVideos.size - pendingIds.length;

  // 新增：全部已下载时提前返回
  if (pendingIds.length === 0) {
    showToast({ type: 'warning', title: '无需下载', message: '所选视频均已下载' });
    clearSelection();
    return;
  }

  const btn = document.querySelector('.btn-primary[onclick="downloadSelected()"]');
  const originalText = btn.textContent;
  btn.textContent = '添加中...';
  btn.disabled = true;

  const data = await fetchWithErrorHandling('/api/task/batch-create', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      video_ids: pendingIds  // ← 只发送未下载的
    })
  }, '添加下载');

  if (data && data.code === 0) {
    const count = data.data?.count || 0;
    // 新增：区分有无跳过的情况
    if (alreadyDownloaded > 0) {
      showToast({ type: 'success', title: '添加成功', message: `已将 ${count} 个视频添加到下载队列（${alreadyDownloaded} 个已下载已跳过）` });
    } else {
      showToast({ type: 'success', title: '添加成功', message: `已将 ${count} 个视频添加到下载队列` });
    }
    // ... 后续操作改用 pendingIds
  }
}
```

**关键变更点**:
- 新增 `pendingIds` 过滤逻辑，从 `_authorVideosData` 检查 `downloaded` 字段
- 新增 `alreadyDownloaded` 计数
- 新增全部已下载的提前返回
- `video_ids` 从 `Array.from(_selectedVideos)` → `pendingIds`
- toast 提示增加跳过数量
- 后续 DOM 操作（行移除、状态更新）都改用 `pendingIds`

## 4. 设置页下载目录：新增文件夹选择功能

### 4.1 后端：ExposedAPI 新增 select_folder 方法

**文件**: `gui.py`

```python
# === 新增（第136行起）===
def select_folder(self) -> dict:
    """选择文件夹对话框，返回选中路径或取消信息"""
    import webview
    import logging
    logger = logging.getLogger(__name__)

    logger.info("[select_folder] 开始调用 create_file_dialog")
    try:
        folder_path = self._window.create_file_dialog(
            dialog_type=webview.FileDialog.FOLDER,
        )
        logger.info(f"[select_folder] create_file_dialog 返回: {folder_path}")
    except Exception as e:
        logger.error(f"[select_folder] 对话框异常: {e}")
        return {"success": False, "message": f"对话框异常: {e}"}

    if folder_path is None:
        logger.info("[select_folder] 用户取消选择")
        return {"success": False, "message": "已取消选择"}

    if isinstance(folder_path, tuple):
        folder_path = folder_path[0]

    logger.info(f"[select_folder] 返回路径: {folder_path}")
    return {"success": True, "path": folder_path}
```

**调用链**: JS `await window.pywebview.api.select_folder()` → pywebview 后台线程执行 → `create_file_dialog(FOLDER)` → 返回 `{"success": True, "path": "C:\\Users\\..."}`

```js
async function selectDownloadFolder() {
  console.log('[selectDownloadFolder] 开始, pywebview:', !!window.pywebview, ...);
  if (window.pywebview && window.pywebview.api && window.pywebview.api.select_folder) {
```

```js
var _cancelledTaskIds = {};

function handleSSETaskProgress(data) {
  var taskId = data.id;

  if (taskId && _cancelledTaskIds[taskId]) {
    if (data.status === 'done' || data.status === 'error') {
      delete _cancelledTaskIds[taskId];
    }
    return;
  }
  // ... 进度跳跃异常检查
  if (pct > oldPct + 50 && data.status !== 'done' && ...) {
    return;
  }
}
```

```js
function cancelDownload(taskId) {
  var task = _activeTasks.find(t => t.id === taskId);
  var videoId = task ? task.video_id : null;

  // 新增：HTTP请求发出前就标记，防止SSE残留消息竞态
  if (typeof _cancelledTaskIds !== 'undefined') {
    _cancelledTaskIds[taskId] = true;
    setTimeout(function() { delete _cancelledTaskIds[taskId]; }, 5000);
  }

  apiFetch('/api/task/cancel', { ... })
    .then(r => r.json())
    .then(data => {
      if (data.code === 0) {
        updateCancelStatus(taskId, videoId);  // 新增 videoId 参数
      } else {
        if (typeof _cancelledTaskIds !== 'undefined') delete _cancelledTaskIds[taskId];
      }
    });
}

function updateCancelStatus(taskId, videoId) {
  _activeTasks = _activeTasks.filter(t => t.id !== taskId);
  State.tasks.remove(taskId);
```
